class CoinFlipper:
    def __init__(self):
        self.coins = list()
        self.win_cond = {'우유':[],'동민':[]}
        self.winner = str()
    def __iter__(self):
        return self
    def __next__(self):
        self.coins.append(np.random.choice(['앞면','뒷면']))
        if self.coins[-2:] == self.win_cond['우유']:
            self.winner = '우유'
            raise StopIteration
        elif self.coins[-2:] == self.win_cond['동민']:
            self.winner = '동민'
            raise StopIteration
        else:
            pass
    def __call__(self,win_cond):
        self.win_cond = win_cond
        for _ in self:
            pass
    def __repr__(self):
        repr_str = '우유의 승리조건: {}\n동민의 승리조건: {}\n동전을 던진결과들: {}\n최종승리자: {}'.format(
            self.win_cond['우유'],
            self.win_cond['동민'],
            self.coins,
            self.winner
        )
        return repr_str

def derivate(f):
    # step1: 함수오브젝트 f는 입력으로 받은상태
    # step2: 함수오브젝트 f를 이용하여 df라는 함수를 정의
    def df(x):
        h=0.000000000001
        return (f(x+h)-f(x))/h
    # step3: 정의된 df를 리턴
    return df
