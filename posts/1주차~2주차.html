그래프와 접선 반환 함수

# x0의 좌표를 넣으면 접선의 점들을 반환
def make_func(x0): # x0좌표넣기
    f = lambda x: x**3 # 함수 정의
    def func(x):
        h = 0.0001
        a = (f(x0+h)-f(x0))/h
        return a*(x-x0) + f(x0)
    return func
x = np.linspace(-5,5,101) #x범위 선
tan_line = make_func(3)
plt.plot(x,x**3,'--')
plt.plot(x,tan_line(x),'--')
